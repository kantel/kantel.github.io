    PVector location;
    PVector velocity;

    Ball() {

        velocity = new PVector(random(-2, 2), random(-2, 2));

    void checkEdges() {

        } else if (location.x < 0) {
            location.x = width;
        }
        
        if (location.y > height) {
            location.y = 0;
        } else if (location.y < 0) {
            location.y = height;
        }
    }

    void display() {
        stroke(0);
        fill(c);
        circle(location.x, location.y, 16);
    }
}
