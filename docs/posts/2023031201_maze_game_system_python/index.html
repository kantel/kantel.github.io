class Sprite(turtle.Turtle):
  
    def __init__(self):
        turtle.Turtle.__init__(self)
        self.penup()
        self.right(270)
        self.speed(0)

  # Kollisionserkennung (Pythagoras)
    def is_collision(self, other):
        a = self.xcor() - other.xcor()
        b = self.ycor() - other.ycor()
        distance = math.sqrt((a**2) + (b**2))
        if distance < 5:
            return True
        else:
            return False

    def destroy(self):
        self.goto(2000, 2000)
        self.hideturtle()
